#### 字符串乘法，输出 2 的 1000 次方

```java
// 使用BigInteger
import java.math.BigInteger;

public class PowerOfTwo {
    public static void main(String[] args) {
        BigInteger base = BigInteger.valueOf(2);
        BigInteger result = base.pow(1000);
        System.out.println(result);
    }
}

// BigInteger底层使用的快速幂算法

public static long fastPow(long base, int exponent) {
    long result = 1;  // 结果初始化为 1
    while (exponent > 0) {  // 当指数大于0时循环
        // 若指数的二进制最后一位为1（即 exponent 是奇数），则将当前 base 乘入结果
        if ((exponent & 1) == 1) {  
            result *= base;
        }
        base *= base;  // base 自乘（相当于计算 base^(2^k)，k为当前轮次）
        exponent >>= 1;  // 指数右移1位（相当于除以2，舍去余数）
    }
    return result;
}

// 字符串乘法
public class PowerOfTwoWithString {
    public static void main(String[] args) {
        String result = "1";
        for (int i = 0; i < 1000; i++) {
            result = multiplyByTwo(result);
        }
        System.out.println("2^1000 = " + result);
    }

    // 字符串模拟乘以2
    public static String multiplyByTwo(String num) {
        StringBuilder sb = new StringBuilder();
        int carry = 0;
        for (int i = num.length() - 1; i >= 0; i--) {
            int digit = num.charAt(i) - '0';
            int product = digit * 2 + carry;
            sb.append(product % 10);
            carry = product / 10;
        }
        if (carry > 0) {
            sb.append(carry);
        }
        return sb.reverse().toString();
    }
}
```
