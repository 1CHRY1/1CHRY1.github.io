    public boolean backtracking(char[][] board) {
        int n = board.length;
        for (int r = 0; r < n; ++ r) {
            for (int c = 0; c < n; ++ c) {
                if (board[r][c] != '.')
                    continue;
                for (char num = '1'; num <= '9'; ++ num) {
                    if (check(board, r, c, num)) {
                        board[r][c] = num;
                        boolean res = backtracking(board);
                        if (res == true)
                            return true;
                        board[r][c] = '.';
                    }
                }
                return false;
            }
        }
        return true;
    }

    private boolean check(char[][] board, int row, int col, char num) {
        int n = board.length;
        // è¡Œ
        for (int i = 0; i < n; ++ i)
            if (board[row][i] == num)
                return false;
        // åˆ—
        for (int i = 0; i < n; ++ i)
            if (board[i][col] == num)
                return false;
        // ä¹å®«æ ¼
        int locationRow = row / 3, locationCol = col / 3;
        for (int i = 3*locationRow; i < 3*(locationRow+1); ++ i)
            for (int j = 3*locationCol; j < 3*(locationCol+1); ++ j)
                if (board[i][j] == num)
                    return false;
        return true;
    }
